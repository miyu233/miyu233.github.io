1059 Prime Factors（25分）

题目链接

题意：给出一个int范围的整数，按照从小到大的顺序输出其分解位质因数的乘法算式。

#include <iostream>
#include <cmath>
using namespace std;
const int maxn = 100001;
int prime[maxn], pnum = 0;
struct factor {
    int x, cnt;
} fac[10];
bool isPrime(int a) {
    if (a == 0 || a == 1) return false;
    for (int i = 2; i <= sqrt(a); i++) {
        if (a % i == 0) return false;
    }
    return true;
}
void findPrime() {
    for (int i = 2; i < maxn; i++) {
        if (isPrime(i)) prime[pnum++] = i;
    }
}
int main() {
    findPrime();
    int n, num = 0;
    scanf("%d", &n);
    if (n == 1) printf("1=1");
    else {
        printf("%d=", n);
        for (int i = 0; i < pnum && prime[i] <= sqrt(n); i++) {
            if (n % prime[i] == 0) {
                fac[num].x = prime[i];
                fac[num].cnt = 0;
                while (n % prime[i] == 0) {
                    fac[num].cnt++;
                    n /= prime[i];
                }
                num++;
            }
            if (n == 1) break;
        }
        if (n != 1) {
            fac[num].x = n;
            fac[num++].cnt = 1;
        }
        for (int i = 0; i < num; i++) {
            if (i > 0) printf("*");
            printf("%d", fac[i].x);
            if (fac[i].cnt > 1) {
                printf("^%d", fac[i].cnt);
            }
        }
    }
    return 0;
}

1052 Linked List Sorting（25分）

题目链接

链表相关题目，大致意思是将结点序列按照key值进行排序，然后依次输出结点的地址，key值和下一个结点的地址，注意排完序后每个结点的next就改变了。

虽然题目案例中没有体现，但是给出的结点有可能不在链表中，所以结构体需要有个flag判断。

最后cmp函数的意思是如果a和b的flag其中有一个是false，则排序flag，否则排序key。这样可以在保证链表中结点按照key值从小到大排序的同时不在链表中的结点全部放在后面。

void postorder(int root, int start, int end) {
    if (start > end) return;
    int i = start;
    while (i < end && in[i] != pre[root]) i++;

1020 Tree Traversals（25分）

题目链接

给出一棵二叉树的后序遍历序列和中序遍历序列，求这棵二叉树的层序遍历序列。

#include <iostream>
#include <vector>
#include <queue>
using namespace std;
const int maxn = 50;
int n, post[maxn], in[maxn];
struct node {
    node* left;
    int data;
    node* right;
};
node* create(int inL, int inR, int postL, int postR) {
    if (postL > postR) return NULL;
    int k;
    node* root = new node;
    root->data = post[postR];
    for (k = 0; k < inR; k++) {
        if (post[postR] == in[k]) break;
    }
    int numLeft = k - 1;
    root->left = create(inL, k - 1, postL, postL + numLeft - 1);
    root->right = create(k + 1, inR, postL + numLeft, postR);
    return root;
}
void BFS(node* root) {
    queue<node*> q;
    q.push(root);
    int num = 0;
    while (!q.empty()) {
        node* now = q.front();
        q.pop();
        printf("%d", now->data);
        num++;
        if (num < n) printf(" ");
        if (now->left != NULL) q.push(now->left);
        if (now->right != NULL) q.push(now->right);
    }
}
int main(){
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &post[i]);
    for (int i = 0; i < n; i++) scanf("%d", &in[i]);
    node* root = create(0, n - 1, 0, n - 1);
    BFS(root);
    return 0;
}

1053 Path of Equal Weight（30分）

题目链接

给定一棵树和每个结点的权值，求所有从根结点到叶子结点的路径，使得每条路径上的结点的权值之和等于给定的常数s。如果有多条这样的路径，则按照路径非递增的顺序输出。

#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
const int maxn = 110;
int n, m, s;
int path[maxn];     // path记录已经走过路径的编号
struct Node {
    int weight;
    vector<int> child;
} node[maxn];
bool cmp(int a, int b) {
    return node[a].weight > node[b].weight;
}
void DFS(int index, int numNode, int sum) {
    // index表示当前访问的结点编号，numNode表示path中的结点数，也代表递归层数，sum表示当前权和
    if (sum > s) return;
    else if (sum == s) {
        if (node[index].child.size() == 0) {
            for (int i = 0; i < numNode; i++) {
                printf("%d", node[path[i]].weight);
                if (i < numNode - 1) printf(" ");
                else printf("\n");
            }
            return;
        } else return;
    } else {
        for (int i = 0; i < node[index].child.size(); i++) {
            int child = node[index].child[i];
            path[numNode] = child;
            DFS(child, numNode + 1, sum + node[child].weight);
        }
    }
}
int main() {
    scanf("%d%d%d", &n, &m, &s);
    for (int i = 0; i < n; i++) scanf("%d", &node[i].weight);
    for (int i = 0; i < m; i++) {
        int id, num, child;
        scanf("%d%d", &id, &num);
        for (int j = 0; j < num; j++) {
            scanf("%d", &child);
            node[id].child.push_back(child);
        }
        // 每次将某结点的子结点全部添加完毕时进行排序
        sort(node[id].child.begin(), node[id].child.end(), cmp);
    }
    path[0] = 0;
    DFS(0, 1, node[0].weight);
    return 0;
}

1043 Is It a Binary Search Tree（25分）

题目链接

==问题描述==：给出n个正整数来作为一棵二叉排序树的结点插入顺序，问：这串序列是否是这棵二叉排序树或者这棵二叉排序树的镜像树的先序序列。所谓镜像树是指交换二叉树的所有结点的左右子树得到的树。如果时镜像树，则输出YES，并输出队赢的树的后序序列，否则，输出NO。

==解题思路==：给定的插入序列，那么写一个二叉排序树的insert函数来建立一棵二叉排序树。因为需要判断二叉排序树的先序和镜像先序，又有可能要输出二叉排序树的后序和镜像后序，所以分别写一个方法来生成对应的序列。镜像树只需要将递归左子树和递归右子树两行代码交换即可。注意如果递归得到的结点是NULL，则应该新建结点，并将其左子树和右子树地址设为NULL。

#include <iostream>
#include <vector>
using namespace std;
struct node {
    int data;
    node* left, right;
};
void insert(node* &root, int data) {
    if (root == NULL) {
        root = new node;    // 遇上空结点即为插入位置
        node->data = data;
        node->left = NULL;
        node->right = NULL;
    } else {
        if (data < node->data) insert(root->left, data);
        else insert(root->right, data);
    }
}
void preorder(node* root, vector<int> &v) {
    if (root == NULL) return;
    v.push_back(root->data);
    preorder(root->left, v);
    preorder(root->right, v);
}
void preorderMirror(node* root, vector<int> &v) {
    if (root == NULL) return;
    v.push_back(root->data);
    preorderMirror(root->right, v);
    preorderMirror(root->left, v);
}
void postorder(node* root, vector<int> &v) {
    if (root == NULL) return;
    postorder(root->left, v);
    postorder(root->right, v);
    v.push_back(root->data, v);
}
void postorderMirror(node* root, vector<int> &v) {
    if (root == NULL) return;
    postorderMirror(root->right, v);
    postorderMirror(root->left, v);
    v.push_back(root->data, v);
}
int main() {
    int n, temp;
    node* root = NULL;
    scanf("%d", &n);
    vector<int> origin, pre, prem, post, postm;
    for (int i = 0; i < n; i++) {
        scanf("%d", &temp);
        origin.push_back(temp);
        insert(root, temp);
    }
    preorder(root, pre);
    preorderMirror(root, prem);
    postorder(root, post);
    postorderMirror(root, postm);
    if (origin == pre) {
        printf("YES\n");
        for (int i = 0; i < n; i++) {
            if (i != 0) printf(" ");
            printf("%d", post[i]);
        }
    } else if (origin == prem) {
        printf("YES\n");
        for (int i = 0; i < n; i++) {
            if (i != 0) printf(" ");
            printf("%d", postm[i]);
        }
    } else printf("NO");
    return 0;
}

1034 Head of a Gang（30分）

题目链接

DFS遍历求犯罪团伙头目姓名和成员人数。

#include <iostream>
#include <vector>
#include <map>
using namespace std;
int n, k, numPerson = 0;
const int maxn = 2001;
int g[maxn][maxn] = {0}, weight[maxn] = {0};
bool vis[maxn] = {false};
map<string, int> strtoi;
map<int, string> itostr;
map<string, int> gang;
void DFS(int nowVisit, int& head, int& numMember, int& totalValue) {
    numMember++;
    vis[nowVisit] = true;
    if (weight[nowVisit] > weight[head]) head = nowVisit;
    for (int i = 0; i < numPerson; i++) {
        if (g[nowVisit][i] > 0) {
            totalValue += g[nowVisit][i];
            g[i][nowVisit] = 0;
            if (vis[i] == false) DFS(i, head, numMember, totalValue);
        }
    }
}
void DFSTravel() {
    for (int i = 0; i < numPerson; i++) {
        if (vis[i] == false) {
            int head = i, numMember = 0, totalValue = 0;
            DFS(i, head, numMember, totalValue);
            if (numMember > 2 && totalValue > k) {
                gang[itostr[head]] = numMember;
            }
        }
    }
}
int change(string s) {
    if(strtoi.find(s) != strtoi.end()) return strtoi[s];
    else {
        strtoi[s] = numPerson;
        itostr[numPerson] = s;
        return numPerson++;
    }
}
int main() {
    cin >> n >> k;
    string s;
    for (int i = 0; i < n; i++) {
        string s1, s2;
        int w, id1, id2;
        cin >> s1 >> s2 >> w;
        id1 = change(s1);
        id2 = change(s2);
        weight[id1] += w;
        weight[id2] += w;
        g[id1][id2] += w;
        g[id2][id1] += w;
    }
    DFSTravel();
    cout << gang.size() << endl;
    for (auto i : gang) {
        cout << i.first << " " << i.second << endl;
    }
    return 0;
}

最短路径

1003 Emergency (25 分)

题目链接

  • 这道题是一道Dijkstra算法练习题，在基础写法的基础上需要求最短路径条数和点权之和最大，因此可以开一个num[ ]来储存每一个顶点的最短路径条数，weight[ ]储存点权值，w[ ]用于储存每个顶点最短路径的点权值和。注意要点有：
    • minn的值不应该是0，而是inf，如果为0则任意一条最短路径都小于等于它
    • 此题是无向边，所以记得写g[u][v] = g[v][u]，如果把这题当成有向边来做，那么测试点2和测试点5会无法通过
    • 代码中vis[u] = true是u不是s
    • 在int v = 0的循环中，如果小于则需要更新最短路径，用num[u]覆盖num[v]，因为最短路径改变；如果等于则说明最短路径条数有多，累加即可
#include <iostream>
#include <algorithm>
using namespace std;
int n, m, s, ed;
const int maxn = 501;   // 最大顶点数
const int inf = 1000000000; // 表示一个很大的树初始化d数组
// 分别表示起点到各顶点的距离，点权，最大权值和，边数和，图
int d[maxn], weight[maxn], w[maxn], num[maxn], g[maxn][maxn];
bool vis[maxn] = {false};     // 判定是否访问过
void Dijkstra(int s) {
    d[s] = 0;
    num[s] = 1;
    w[s] = weight[s];
    for (int i = 0; i < n; i++) {
        // 记住这个地方min是无穷大，不是0！！！！
        int u = -1, minn = inf;   // u表示当前未访问过顶点中距离s最小的那个
        for (int j = 0; j < n; j++) {
            if (vis[j] == false && d[j] < minn) {
                u = j;
                minn = d[j];
            }
        }
        if (u == -1) return;    // 说明起点s不存在相连的边
        vis[u] = true;
        // 这一步是遍历所有与中介点相连的边，观察能否通过中介点来降低最短路径长度
        // 如果可以就更新
        for (int v = 0; v < n; v++) {
            if (vis[v] == false && g[u][v] != inf) {
                if (g[u][v] + d[u] < d[v]) {
                    d[v] = d[u] + g[u][v];
                    num[v] = num[u];
                    w[v] = weight[v] + w[u];    
                } else if (g[u][v] + d[u] == d[v]) {
                    if (w[u] + weight[v] > w[v]) {
                        w[v] = weight[v] + w[u];
                    }
                    // 这行代码的意思是v点除了原来的条数还有从num[u]过来的新条数，累加
                    num[v] += num[u]; // 注意最短路径条数和点权无关，必须写在外面
                }
            }
        }
    }
}
int main() {
    scanf("%d%d%d%d", &n, &m, &s, &ed);
    for (int i = 0; i < n; i++) scanf("%d", &weight[i]);
    fill(g[0], g[0] + maxn * maxn, inf); // 初始化图
    fill(d, d + maxn, inf); // 初始最短路径置为inf
    fill(num, num + maxn, 0);
    fill(w, w + maxn, 0);
    for (int i = 0; i < m; i++) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);
        g[u][v] = g[v][u] = w;
    }
    Dijkstra(s);
    printf("%d %d\n", num[ed], w[ed]);
    return 0;
}
